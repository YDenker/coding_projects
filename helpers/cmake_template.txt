# CMakeLists.txt 

file(GLOB SOURCES "src/*.cpp")
file(GLOB HEADERS "include/*.h")

add_executable (_name_ ${SOURCES} ${HEADERS})

target_include_directories(_name_ PRIVATE include)


if (CMAKE_VERSION VERSION_GREATER 3.12)
  set_property(TARGET _name_ PROPERTY CXX_STANDARD 20)
endif()

set_target_properties(_name_ PROPERTIES OUTPUT_NAME "_name_")
